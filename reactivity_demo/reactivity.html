# 获取变化通知

在A发生变化的时候，依赖A的B自动更新。 那我们怎么知道数据发生变化？

在Ruby中，一个属性的getter 跟 setter可以这样定义：

class A
  def a
    @a
  end

  def a=(value)
    p '大家请注意：我发生变化了...'
    @a = value
  end
end

demo = A.new
demo.a = 100
# "大家请注意：我发生变化了..."
# => 100

这里的setter里 console.log ==> 相当于一个callback。 实现watch功能：


let expense = {
  duration: 2,    // 停车时长
  day_money: 100, // 一天多少钱
}

function watch(obj, key, callback) {
  let value = obj[key]
  Object.defineProperty(obj, key, {
    enumerable: true,
    configurable: true,
    get() {
      return value
    },
    set(newValue) {
      if(newValue === value) {
        return
      } else {
        let oldValue = value
        value = newValue
        callback(newValue, oldValue)  // setter 中触发回调函数，并将新值与旧值作为参数传入
      }
    }
  })
}

watch(expense, 'duration', (newValue, oldValue) => {
  console.log(`duration属性被修改了，新值为：${newValue}，旧值为：${oldValue}`)
})

expense.duration = 99
// duration属性被修改了，新值为：99，旧值为：2
// 99

# 通知变化

我们的目标是：A变化时，依赖于A的元素也跟着变化。 使用发布/订阅的模式，有没有很熟悉~ 在 duration 发生变化时，发布”我已经更新了”的通知，所有订阅者各自根据注册的回调进行处理（打日志，重新渲染，重新计算等）

class Dep {
  constructor() {
    // 订阅者列表
    this.subs = []
  }

  // 添加订阅
  addSub(sub) {
    this.subs.push(sub)
  }

  // 触发订阅
  notify(newValue, oldValue) {
    this.subs.forEach(sub => {
      sub.update(newValue, oldValue)
    })
  }
}

基本的套路。这里我们注意一下，每个订阅者都要能够响应update方法，这里就有一定局限，得是对象才行。如果此时订阅者是个匿名函数，就处理不了。所以这里可以抽出个中间层出来 ----> 叫做 Watcher。

Watcher 类有个 update 方法，注册订阅的时候，callback 对应到 update 即可。

// Watcher类，每个 Watcher 为一个订阅源
class Watcher {
  constructor(callback) {
    this.callback = callback
  }

  update(newValue, oldValue) {
    // 被触发后执行回调
    this.callback(newValue, oldValue)
  }
}

let dep = new Dep()
let sub1 = new Watcher((newValue, oldValue) => {
  console.log('第一条......')
})

let sub2 = new Watcher((newValue, oldValue) => {
  console.log('第二条......')
})

// 一个watcher 相当于 一个sub
dep.addSub(sub1)
dep.addSub(sub2)
dep.notify(1, 2)

# 依赖收集

那我们怎么知道属性有哪些订阅者？在哪里进行订阅？

也许可以:

  • 谁依赖(使用)了该属性，谁就是订阅者。
  • 在使用属性值时，会触发getter，在那里进行订阅。

function defineReactive(obj, key) {
  let dep = new Dep() // <========= 为每个键都创建一个 dep
  let value = obj[key]

  // 由于闭包，属性时刻可以访问到以上变量
  Object.defineProperty(obj, key, {
    enumerable: true,
    configurable: true,
    get() {
      dep.depend() // <========= 进行订阅 ????
      return value
    },
    set(newValue) {
      if (newValue == value) {
        return
      } else {
        let oldValue = value
        value = newValue
        dep.notify(newValue, oldValue)   // <========= 变动时触发通知
      }
    }
  })
}

页面初始化的时候，init肯定会执行一次。这个时候所有的页面渲染需要的数据，都会被访问到。而那些无关的数据，就算它们发生改变，也不需要重新渲染。

在页面初始化时，在一个公共的地方存放当前watcher(callback就是更新页面处理), 当到依赖数据时，就将当前watcher放到对应的dep里面。这是一个很巧妙的做法。

遍历对象，定义属性描述：

// 使一个对象响应化
// 遍历对象所有key，全部设置属性描述
function observify(obj= {}) {
  Object.keys(obj).forEach((key) => {
    defineReactive(obj, key)
  })
}

function defineReactive(obj, key) {
  let dep = new Dep() // <========= 为每个键都创建一个 dep
  let value = obj[key]

  // 由于闭包，属性时刻可以访问到以上变量
  Object.defineProperty(obj, key, {
    enumerable: true,
    configurable: true,
    get() {
      // 相当于一个状态值。用来记录当前线程正在处理哪个watcher,如果存在，将其加入到订阅列表中
      if (Dep.target) {
        dep.depend() // <========= 进行订阅
      }

      return value
    },
    set(newValue) {
      if (newValue == value) {
        return
      } else {
        let oldValue = value
        value = newValue
        dep.notify(newValue, oldValue)   // <========= 变动时触发通知
      }
    }
  })
}

Dep代码：

let uid = 1

class Dep {
  constructor() {
    // 为每个 dep 标记一个 uid
    this.id = uid++

    // 订阅者列表
    this.subs = []
  }

  // 添加订阅
  addSub(sub) {
    this.subs.push(sub)
  }

  // 触发订阅
  notify(newValue, oldValue) {
    this.subs.forEach(sub => {
      sub.update(newValue, oldValue)
    })
  }

  depend() {
    // 这里的做法是 将当前的dep传出去。
    // 而不是将watcher传进来，这一点很重要
    // Dep.target 是个watcher
    Dep.target && Dep.target.addDep(this)
  }

}

中间层的Watcher:

// 用来记录当前是哪个watcher, 共用。
// 将当前watcher放入当前属性的订阅列表中
Dep.target = null

// Watcher类，每个 watcher 为一个订阅源
class Watcher {
  constructor(expFn, callback) {
    this.expFn = expFn
    this.callback = callback
    this.depIds = new Set() // 标记当前 watcher 已经加入到了哪些 dep
    this.value = this.subAndGetValue()
  }

  update(newValue, oldValue) {
    // 被触发后执行回调
    this.callback(newValue, oldValue)
  }

  // 这里才将watcher添加进dep
  addDep(dep) {
    //防止重复订阅，防止在一个 dep 中订阅两次
    if (!this.depIds.has(dep.id)) {
      this.depIds.add(dep.id)
      dep.addSub(this)
    }
  }

  // 执行依赖收集函数，订阅依赖！
  // 只在expFn里面访问到属性，才会注册相关订阅
  // 为什么要定义为expFn？这里执行函数的内容，比如 a + b.
  // 访问 a 时，就会进入到 a的getter, 这个时候,
  // 发现Dep.target有值 => 那么执行 dep.depend() => addDep()实现了真正的订阅
  // 然后将Dep.target又置为空。
  // b的操作跟a类似。
  // 所以在 expFn 访问过的属性（1个 或 多个），都实现了订阅。
  subAndGetValue() {
    Dep.target = this
    let _value = this.expFn.call()
    Dep.target = null

    return _value
  }
}

// 封装成一个 Watcher
function watch(expFn, callback) {
  return new Watcher(expFn, callback)
}
